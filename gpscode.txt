void getLocation() {
    HTTPClient http;
    String url = "https://positioning.hereapi.com/v2/locate?apiKey=" + String(HERE_API_KEY);
    String jsonPayload = "{ \"wlan\": [";

    int numNetworks = WiFi.scanNetworks();
    if (numNetworks <= 0) {
        Serial.println("No WiFi networks detected. Cannot determine location.");
        return;
    }

    int networksToSend = min(numNetworks, 5);
    for (int i = 0; i < networksToSend; i++) {
        jsonPayload += "{ \"mac\": \"" + WiFi.BSSIDstr(i) + "\" }";
        if (i < networksToSend - 1) jsonPayload += ", ";
    }
    jsonPayload += "] }";

    http.begin(url);
    http.addHeader("Content-Type", "application/json");
    int httpResponseCode = http.POST(jsonPayload);

    if (httpResponseCode > 0) {
        String response = http.getString();
        Serial.println("Location Data: " + response);

        int latIndex = response.indexOf("\"lat\":");
        int lonIndex = response.indexOf("\"lng\":");
        if (latIndex != -1 && lonIndex != -1) {
            latitude = response.substring(latIndex + 6, response.indexOf(',', latIndex)).c_str();
            longitude = response.substring(lonIndex + 6, response.indexOf('}', lonIndex)).c_str();
        } else {
            Serial.println("Error parsing latitude and longitude.");
        }
    } else {
        Serial.print("HTTP request failed, response code: ");
        Serial.println(httpResponseCode);
    }
    http.end();
}

void setup() {

    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED) {
        delay(1000);

    }
    Serial.println("\nConnected to WiFi!");

    server.on("/", handleRoot);
    server.on("/location", handleLocation);
    server.begin();
}

void loop() {
    server.handleClient();

    static unsigned long lastUpdate = 0;
    if (millis() - lastUpdate > 30000) {
        getLocation();

}
